    def initialize(argv=nil)
      @argv ||= ARGV.dup

      @noharm = @argv.delete('--dryrun') || @argv.delete('--noharm')
      @debug  = @argv.delete('--debug')

      @argv.reject!{ |e| e =~ /^-/ }

      @location = @argv.shift || '.'
      #@output = @argv.shift
    end
