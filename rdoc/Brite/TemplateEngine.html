    def render_format(format, text)
      case format
      when 'rdoc'
        rdoc(text)
      when 'markdown'
        rdiscount(text)
      when 'textile'
        redcloth(text)
      when 'haml'
        haml(text)
      when /^coderay/
        coderay(text, format)
      else # html
        text
      end
    end

    def render_stencil(stencil, text, attributes)
      case stencil

      when 'liquid'
        liquid(text, attributes)

        text
