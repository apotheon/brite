    def sort_files
      files = Dir['**/*']
      files.each do |file|
        temp = false
        name = File.basename(file)
        ext  = File.extname(file)
        case ext
        when '.layout'
          layouts << Layout.new(self, file)
        when '.page' #*%w{.markdown .rdoc .textile .whtml}
          pages << Page.new(self, file)
        when '.post'
          posts << Post.new(self, file)
        end
      end
      posts.sort!{ |a,b| b.date <=> a.date }
    end
