def configure
  if file = Dir[File.join(location, CONFIG_FILE)].first
    data = YAML.load(File.new(file))
    data.each do |k,v|
      __send__("#{k}=", v)
    end
  end
end

def pom=(set)
  return unless set
  require 'pom'
  Brite::Context.class_eval do
    def project
      @project ||= POM::Project.new
    end
  end
end
